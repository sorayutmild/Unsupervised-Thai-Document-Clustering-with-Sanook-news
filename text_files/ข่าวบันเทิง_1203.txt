ประเด็นแซ่บๆ ตั้งแต่เริ่มต้นศักราช กรณีคลิปหลุดเหตุทะเลาะวิวาทที่มีดาราสาว "แพท ณปภา" เข้าไปมีส่วนเกี่ยวข้องอยู่ด้วย เรื่องราวฉาวโฉ่ยังไม่สงบลง เพราะประเด็นถูกบิดไปถึงเรื่อง "ผู้ชายและความรัก" เนื่องจากหนุ่มคนหนึ่งที่เคยพบหากับสาวแพท ได้บอกเลิกและกลับไปคบกับแฟนเก่า ซึ่งเป็นคู่กรณีที่ถูกรุมทำร้ายในคลิปฉาวดังกล่าว
ล่าสุด แพท ณปภา ได้ไปชี้แจงผ่านทางรายการ เจาะข่าวเด่น สรยุทธ ทางช่อง 3 พร้อมกับเล่าถึงประเด็นต่างๆ ที่เกิดขึ้นกับฝ่ายตัวเอง ขณะที่อีกมุมยังมีคู่หนุ่มสาวที่เป็นคู่กรณีจากเรื่องนี้ นั่นก็คือ สาวเบญ กับ หนุ่มแจ๊ค ที่ออกมาชี้แจงสิ่งที่เกิดขึ้นกับฝ่ายตัวเองเช่นเดียวกัน
เบญ - เริ่มเรื่องเลยก็คือวันนั้นคนที่ชื่อ "นาฟ" หลานสาวของแพท เขาโทรมาถามว่า "หนูเป็นอะไรกับแจ็ค" หนูก็เลยถามกลับไปว่า "ทำไม" เขาก็บอกว่า "ไม่รู้เหรอว่าแจ็ครู้จักกับพี่สาวเขา" หนูก็เลยตอบกลับไปว่า "จริงเหรอ ไหนแจ๊คบอกว่าเลิกกันแล้ว" จากนั้นเขาก็เริ่มขึ้นเสียงและก็เกิดการเถียงกันไปมา และตัวเจ้าของเรื่องเองเขาก็บอกกับหนูว่า "ถ้าหากน้องเขาพูดอะไรไม่ดีไป เขาก็ต้องขอโทษด้วย"
และหลังจากนั้นพอคุยจบหนูก็เข้าไปโพสต์ข้อความในเฟซบุ๊คตัวเองว่า "เราไม่ได้ยุ่งกับใคร ใครก็อย่ามายุ่งกับเรา" ซึ่งหนูไม่ได้แท็กเขาเลย แต่อยู่ดีๆ เขาก็ไปโพสต์ในเฟซบุ๊กของเขาเองว่า "มึงไม่อยากยุ่งกับใคร ใครก็ไม่อยากยุ่งกับมึงเหมือนกัน" จากนั้นก็มีคนถามว่าเขาด่าใคร เขาก็เลยแคปเอาภาพเฟซบุ๊กหนูไปแปะแล้วก็บอกว่า "ด่าอีด-กนี่"
ซึ่งไปๆ มาๆ พอรุ่นพี่หนูเข้าไปเห็นข้อความเขาก็เลยโพสต์ถามว่า "ด่ากันขนาดนี้เลยไม่แรงไปเหรอ" แล้วฝั่งนั้นก็สวนกลับมา "ทำไม แล้วมึงจะออกตัวเหรอ" พอหนูเห็นเข้าหนูก็เลยไปโพสต์ข้อความว่า "นี่ร้อนตัวหรือไง หนูก็โพสต์หน้าเฟซบุ๊กหนูปกติไม่ได้ไปแท็กอะไรเธอสักหน่อย"
จากนั้นมันก็เลยมีการโทรคุยโทรเคลียร์อีกรอบ และนัดเจอกันประมาณ 4-5 โมง ซึ่งพอไปถึงที่เกิดเหตุหนูเห็นเลยว่าฝั่งเขามากันเยอะมากประมาณ 30-40 คน แต่ฝั่งหนูมากันแค่ 7-8 คนเอง"
วันนั้นเราตั้งใจจะเคลียร์แต่ไม่ได้ตั้งใจไปทะเลาะวิวาท ?เบญ - "ค่ะ ถ้าหากเขาคุยกับหนูก่อนหนูก็คุยค่ะ"
ก่อนเกิดเหตุการณ์วันนั้นเราได้คุยกันบ้างไหม ?เบญ - "ตอนแรกที่หนูไปถึงอีกฝั่งหนึ่งเขาก็เรียกหนูบอกว่า "เข้ามาสิ เข้ามาคุยข้างในสิ" ซึ่งพอหนูเดินเข้าไปคนเดียว เขาก็ปิดประตู และหนูคุยกับพี่เขาได้แป๊ปนึง หลานสาวเขาก็กระโดดใส่หนูก่อนเลย พอกระโดดใส่ปุ๊ป เราก็เหมือนกับว่าตบกัน วินาทีนั้นหนูรู้สึกเลยว่าหลายมือมากที่เข้ามาจิกหัว"
"พอแม่หนูเห็นภาพแบบนั้นเขาก็เลยเปิดประตูผลักเข้าไปเพื่อจะดึงลูกออก แต่มือแม่หนูไปโดนที่ปิดหน้าเขา(แพท)จนหลุด เรื่องมันก็เลยยิ่งไปกันใหญ่ จากนั้นพอพี่สะใภ้หนูเห็นท่าไม่ดีเขาก็เข้าไปลากหนูออกมาและบอกให้หนูขึ้นรถกลับไปก่อน"
เราบาดเจ็บส่วนไหนบ้างจากเหตุการณ์นั้น ?เบญ - "หน้าอย่างเดียวเลยค่ะ อาการฟกช้ำตามตัวไม่มี"
เห็นว่าตอนนี้ทางผู้ปกครองของเราได้เข้าไปแจ้งความลงบันทึกประจำวันแล้วด้วย และเราจะมีโอกาสเดินทางไปด้วยตัวเองไหม ?เบญ - "ถ้าเกิดว่าต้องคุยก็ไปค่ะ"
คิดว่าจะมีการแจ้งความเรื่องทำร้ายร่างกายไหม ?เบญ - "ไม่เลยค่ะ ไม่ได้คิดจะแจ้งความหรือเรียกร้องอะไรเลย หนูแค่อยากให้ทุกคนรู้ความจริงเฉยๆ ว่ามันเป็นยังไง หนูโดนคนรุมด่าขนาดนี้ทั้งๆ ที่ไม่ได้ทำผิดอะไรเลย ทำไมถึงต้องมาด่าด้วยคุณเองก็ไม่ได้อยู่ในเหตุการณ์"
สำหรับคลิปที่ออกมาฝั่งเราเป็นคนเผยแพร่จริงไหม ?เบญ - "คลิปนั้นหนูได้จากรุ่นน้องค่ะ แต่ตอนแรกหนูไม่ได้ลง ฝั่งนั้นเป็นคนลงก่อน หลักฐานหนูก็มี มีเวลาชัดเจน แต่เขาลงเป็นคลิปตัดต่อ ซึ่งมันทำให้หนูโดนว่าหนูก็เลยลงคลิปเองค่ะ เป็นคลิปเต็มและมีคนแชร์เยอะมาก"
จริงๆ เราไม่ได้อยากจะเป็นประเด็นอะไรกับแพท ?เบญ - "หนูไม่ได้อยากอะไรเลยค่ะ เพราะหลังจากที่ตบกันเสร็จเขาก็โทรมาคุยกับหนูบอกว่า "ให้จบกันไปเถอะนะเบญ ถ้าหากมีใครพูดว่าฝั่งเราไปแจ้งความหรืออะไรก็ไม่ต้องไปฟังเพราะต่างคนต่างก็พูดกันไป" ซึ่งตอนแรกตัวหนูเองคือยังไงก็ได้จบก็จบ แต่พอสุดท้ายเรื่องมันเป็นข่าวเขากลับบอกว่าเขาไปแจ้งความเอาไว้ก่อนที่จะมาแล้ว"
ตอนนั้นหลังจากที่เราเห็นบทสัมภาษณ์ของเขากับสื่อเรารู้สึกยังไงบ้าง ?เบญ - "มันคนละเรื่องเลยค่ะ และที่หนูออกมาวันนี้หนูก็แค่อยากให้คนรู้ว่าความจริงมันเป็นยังไง หนูไม่ได้อยากจะเป็นประเด็นที่ต้องโดนคนในสังคมด่าขนาดนี้เลยค่ะ"
เหมือนตอนนี้คนก็มองว่ากลายเป็นเรื่องผู้ชายไปแล้ว ?เบญ - "หนูไม่ได้อยากจะมีปัญหาเรื่องผู้ชายเลยค่ะ ไร้สาระมาก"
และที่เขาสงสัยว่าเรากับแจ๊คไปหลอกเขาอันนี้จริงไหม ?เบญ - "หนูเพิ่งจะกลับมาเจอกับแจ็คได้อาทิตย์สองอาทิตย์เอง เวลาที่เราอยู่ด้วยกันยังแทบจะไม่มีเลย เจอกันครั้งสองครั้งเองค่ะ"

เมื่อผู้สื่อข่าวสอบถามกับ หนุ่มแจ๊ค ที่เป็นคู่กรณีที่เป็นต้นตอของเรื่องราวทั้งหมด เจ้าตัวได้เปิดเผยเกี่ยวกับความสัมพันธ์ระหว่าง สาวเบญ และ สาวแพท อย่างตรงไปตรงมาว่า...
"ผมเลิกกับเบญได้ประมาณ 1 ปีกว่าๆ หลังจากเลิกกับเบญก็คบกับคนอื่นบ้าง ก่อนจะมาเจอกับพี่แพท ผมรับประกันเลยว่า ไม่ได้คบซ้อนแน่นอน ผมกับพี่แพทเราเคลียร์ตัวเองหมดแล้วว่า เราเลิกกันแล้วนะ หลังจากนั้น 2 อาทิตย์ เขาก็หายไปเลย"
"ผมก็ไม่ได้ติดต่ออะไรกับเขาเลย ถึงได้กลับไปคืนดีกับเบญ ไม่ได้คบกับพี่แพทอยู่แล้วไปคบซ้อนกับเบญ ที่มีข่าวออกมาว่าผมไปหลอกเขา คือทุกอย่างรองเท้า เสื้อผ้า เขาซื้อให้ผมจริงๆ แต่บางอย่างผมไม่ได้ต้องการอะไรจากเขาเลย"
"ที่ผมเข้าไปอยู่กับเขาไปทำโน่นนี่ให้ เขาคิดว่าผมไปทำดีเพื่อหวังผลตอบแทน เขามองผมแบบนี้ แต่ผมไปด้วยใจจริงไม่ได้อยากได้อะไรจากเขามากมายเลย ผมพูด ณ ตรงนี้ได้เลยว่าของทุกอย่างที่ได้มาผมให้คืนได้หมดเลย ไม่ได้อยากได้เลย ผมไปด้วยใจจริงไม่ได้ต้องการสิ่งของตอบแทนอะไรเลย"
สถานะที่คบกับแพทเรียกว่าแฟนได้ไหม?"ก็น่าจะคบแบบแฟนนะ สำหรับตัวผม ถามว่าเรารู้จักกันได้ยังไง เริ่มจากผมไปบริจาคของให้คนจรจัดข้างถนนกับเพื่อนๆ ก็เจอกับเขาก็ไม่ได้อะไร รู้ว่าเขาเป็นดารา จากนั้นมีโทรศัพท์เข้ามาบอกว่ามีคนอยากคุยด้วย เขาก็บอกว่าเขาเป็นพี่แพท ตอนแรกยังไม่เชื่อว่าพี่แพทจะไปมาคุยกับผมเรื่องอะไร"
"เขาบอกว่าเขาอยากคบกับผม ผมก็ปฏิเสธเขาไปหลายรอบ บอกผมไม่กล้าคบกับพี่หรอก ตอนนั้นผมยังคิดว่าคนอื่นให้เขามาแกล้งผมเลย เราเริ่มคุยกับเมื่อปลายเดือนมกราคมผ่านทางแชท แล้วเริ่มคุยกันจริงจังช่วงเดือนพฤษภาคม รวมระยะเวลาที่รู้จักกันมาก็ประมาณ 7-8 เดือน"
"ผมก็ไม่รู้ว่าเขาจะจริงจังกับผมขนาดนี้ เรามีเรื่องทะเลาะกันบ่อย เช่น ผมอยากออกไปหาเพื่อน เขาจะรับไม่ได้ที่ผมติดเพื่อน เพราะผมจะลืมทุกอย่าง ลืมโทรบอกว่าจะกลับกี่โมง ทุกครั้งที่ออกไปหาเพื่อนก็จะเหตุให้ทะเลาะกันตลอด แล้วเขาจะเป็นฝ่ายบอกเลิกผมก่อนตลอด"
"ตอนนั้นผมไม่อยากเลิกกับเขา ผมแค่ติดเพื่อน ก็ตามง้อเขาทุกรอบ จนครั้งล่าสุดนี้ผมเป็นคนบอกเลิกเขาก่อน เพราะมันมีเรื่องคนมาดูถูกผม เวลามีคนมาถามพี่แพทว่า ผมเป็นใคร พอเขาบอกว่าเป็นแฟน ผู้ใหญ่ก็จะบอกว่าทำไมไม่เลือกบ้าง ทำไมเอาเด็กแบบนี้มาเป็นแฟน ผมก็เก็บเอาไปคิดว่ามันคงไม่ใช่แล้ว นานๆ ไปคนคงคิดว่าผมเป็นเด็กที่ไหนทำไมมาคบกับดาราดังขนาดนี้ได้"
"นั่นจึงเป็นเหตุผลที่ผมก็ตัดสินใจบอกเลิกกับเขาไป ไม่คิดว่าเรื่องมันจะบานปลายขนาดนี้เลย ตอนที่บอกเลิกกันเราต่างก็อวยพรให้เจอคนที่ดี ผมขอบคุณพี่มากที่ให้อะไรหลายๆ อย่างให้ผมได้เจอสิ่งใหม่ๆ หลายๆ อย่าง ไม่คิดว่าจะมาเกิดเรื่องอย่างนี้"
วันเกิดเรื่องเราอยู่ในเหตุการณ์ด้วยไหม?"ไม่อยู่ครับ ผมอยู่บ้านก็ทราบเรื่องจากเพื่อน เขาโทรมาบอกว่า เบญโดนรุมเละเลย คือรู้ก่อนแล้วว่า เบญกับหลานพี่แพทเขาจะไปมีเรื่องกัน แต่พี่สาวเขาไม่ให้ผมไปด้วย เขากลัวว่าจะเป็นเรื่องผู้ชาย หลังจากเกิดเรื่องฝ่ายนั้นก็ไม่มีใครติดต่อมาเลยเราก็ไม่ได้ติดต่อมาเลย ผมคิดว่าเป็นเรื่องของผู้หญิงไม่ใช่เรื่องของผม แล้วก็มีประเด็นมาว่าผมเป็นแฟนกับนาฟหลานพี่แพทแล้วพยายามไปหลอกลวงหลานเขา เอาเงินมาเลี้ยงเบญ ผมงงมากทำไมออกข่าวไปแบบนั้น"
เบญเคยเรียกร้องอะไรจากเราไหม?"หลังจากกลับมาคบกันเคยเจอกันแค่ 2 ครั้ง พูดตรงๆ ครั้งแรกที่เจอกันเบญยังเลี้ยงข้าวผมเลย ถามว่าเสียความรู้สึกใหมที่ข่าวออกไปแบบนี้ ก็เสียความรู้สึกครับเขาทำให้ผมดูแย่ว่าผมไปหลอกเงินเขามา ทำให้ผมสองคนดูไม่ดีไปหลอกเอาเงินเขามา"
"ทั้งที่ตอนเลิกกับเขาเงินสักบาทผมยังไม่มีติดกระเป๋าเลย ต้องเอาเงินในธนาคารมาใช้ นอกจากเสื้อผ้าที่เขาซื้อให้ก็มีแหวนทองกับนาฬิกา ไม่มีเงินสดเลย ตอนคบกันก็มีไปไหนมาไหนด้วยกันปกติ แต่ไม่ได้เดินจับมือหรือเดินใกล้ชิดกัน ผมจะเดินตามหลังตลอดผมไม่อยากเปิดตัวอะไรมาก คนแถวบ้านก็มองออกมาเราเป็นแฟนกันเพราะไปไหนด้วยกันตลอด"
เราอยู่ในหมู่บ้านเดียวกันถึงได้รู้จักกันในเฟซบุ๊กหรือว่ายังไง?เบญ - "คือเราก็มีเพื่อนกันในเฟซบุ๊คค่ะแต่เขาก็ไม่รู้หรอกว่าเราคือใคร"
จริงๆ แล้วเรื่องนี้เราอยากให้จบยังไง ?เบญ - "หนูก็อยากให้ทุกคนรู้ว่าความจริงมันเป็นยังไง ทำไมถึงต้องมารุมด่ากันขนาดนี้ เราก็ไม่เคยรู้จักไม่เคยอะไรกันมาก่อน"
คิดว่าจะมีการเจรจาให้เรื่องจบลงด้วยดีได้ไหม ?เบญ - "ต้องดูที่ฝั่งเขาด้วยค่ะว่าเขาจะเอายังไง"
สำหรับแจ็คเองคิดว่าจะหาโอกาสเข้าไปคุยกับพี่แพทให้เรื่องมันเงียบลงไหม ?แจ็ค - "ถ้าหากมีโอกาสได้คุยผมก็อยากจะคุยกับพี่เขาครับ แต่ว่าตอนนี้ยังไม่ได้ติดต่อไปเลย"
ถ้าพี่แพทดูอยู่ตอนนี้เราอยากจะบอกอะไรเขา ?แจ็ค - "ผมก็ไม่รู้จะพูดยังไงดี เพราะถ้าพี่เขาพูดให้ผมกับเบญดูในทางลบผมก็ไม่รู้ว่ามันจะเป็นยังไง แต่ผมอยากให้พี่เขาพูดความจริงจะดีกว่าว่าเรื่องมันเป็นยังไง และที่ผมไม่ได้พูดออกสื่อออกเฟซก็เพราะผมอยากให้เงียบ ผมเห็นใจเขาเพราะถ้าผมออกมาพูดเดี๋ยวเขาจะดูเสื่อมเสีย ผมยังไม่พูดเลยว่าผมเป็นแฟนเขา ผมไม่เคยเปิดเผยว่าผมเคยคบกับเขา แต่ก็มีคนไปขุดมาเรื่อยๆ ซึ่งผมไม่รู้ว่าใครจนสุดท้ายเรื่องมันก็เลยเป็นข่าว และพอมันมีชื่อผมแบบนี้ผมก็ต้องออกมาพูดครับ"
