ผลตรวจเนื้อเยื่อทางนิติวิทยาศาสตร์ ของบาดแผลบนร่างกาย แตงโม จากหน่วยงานภายนอกออกแล้ว แผลที่ต้นขาขวาไม่ใช่แผลที่ทำให้เสียชีวิต
หลังแพทย์นิติเวช รพ.ตำรวจ ได้มีการส่งเนื้อเยื่อบาดแผลบนร่างกาย แตงโม ไปให้กับแพทย์จากหน่วยงานภายนอกร่วมตรวจสอบด้วย
ล่าสุดวันนี้(13 มี.ค. 65) อีจัน ได้รับรายงานว่า หน่วยงานแห่งหนึ่งซึ่งเป็นหน่วยงานภายนอกที่สถาบันนิติเวชวิทยา รพ.ตำรวจ ส่งเนื้อเยื่อของบาดแผลบนร่างกายแตงโมไปตรวจสอบเพื่อยืนยันผล การตรวจพิสูจน์
ผู้เชี่ยวชาญในหน่วยงานดังกล่าว ได้ส่งผลการตรวจด้วยกล้องจุลทรรศน์ ที่เป็นอุปกรณ์สำหรับส่องดูวัตถุที่มีขนาดเล็กเกินกว่าการมองเห็นด้วยตาเปล่า แจ้งว่า จากการวิเคราะห์ตรวจสอบพบว่าสภาพเนื้อเยื่อจากบาดแผลที่ใหญ่ที่สุดบริเวณต้นขาขวานั้น เป็นบาดแผลที่เกิดก่อนเสียชีวิต
จากการตรวจพบการเปลี่ยนแปลงแสดงผลกระทบ แสดงอาการ พบว่ามีเลือดออกที่ขอบแผล สภาพบาดแผลไม่ตัดกับเส้นเลือด ซึ่งไม่ได้เป็นบาดแผลที่ทำให้เกิดการเสียชีวิต แต่การเสียชีวิตนั้น แพทย์ผู้ชันสูตรจะทราบดีว่าเกิดจากอะไร
ทั้งนี้ บาดแผลที่เกิดขึ้นหลังเสียชีวิต แตกต่างจากสภาพบาดแผลดังกล่าว คือ ตรวจสอบผ่านกล้องจุลทรรศน์จะพบว่าไม่มีเลือดออก เนื้อเยื่อไม่มีฉีกขาด บาดแผลไม่ถูกทำลาย
